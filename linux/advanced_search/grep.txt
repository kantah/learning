grep -l "<строка>" <директория>  # список файлов, где есть <строка>
grep -L "<строка>" <директория>  # список файлов, в которых строки нет

grep -n "<строка>" <директория>  # выводить номер строки в файле
grep -m <число> "<строка>" <директория>  # не искать далбше заданного числа вхождений

grep -A <число> "<строка>" <путь>  # выводить <число> строк после вхождения
grep -B <число> "<строка>" <путь>  # выводить <число> строк перед вхождением
grep -C <число> "<строка>" <путь>  # выводить <число> строк вокруг вхождения

grep -E "<шаблон>" <файл>  # найти шаблон в файле
grep -E "^go" <файл>  # найти строки, начинающиеся с "go"
grep -E "go$" <файл>  # найти строки, заканчивающиеся на "go"
grep -E "[a-z]+right" <файл>  # слова из 5 и более букв, заканчивающиеся на "right"
grep -E "[a-z]?right" <файл>  # слова из 4-5 букв, заканчивающиеся на "right"
grep -E "[a-zA-Z]*ight" <файл>  # слова, заканчивающиеся на "ight"(разрешены заглавные буквы)
