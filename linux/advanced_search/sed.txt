cat <файл> | sed '<инструкция>'
sed '<инструкция>' <файл>  # потоковый редактор: читает строки из stdin или из файла, обрабатывает их по инструкции и выводит результат в stdout

Если нужно вывести в файл:
><файл>  # обычное перенаправление
-i, --in-place  # перезаписать входящий файл

sed 's/John/Nick/g' old.txt>new.txt  # заменить все John на Ник
sed -r 's/J[a-z]*n/Nick/g' old.txt>new.txt  # заменить все слова, начинающиеся на J и заканчивающиеся на n, на Nick

sed -n '2,4p' file.txt  # вывсети строки со 2 по 4
sed '2,4d' file.txt  # вывести все строки кроме 2-4
